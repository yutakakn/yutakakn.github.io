#include <stdio.h>
#include <string.h>

#define MIN 1000
#define MAX 4700
#define KETA 10
#define N ((KETA-1)/4 + 1)

/* ‘å’¬”ƒ`ƒFƒbƒN */
int check(int *s)
{
	static int use[10];
	int i,j,t,n;

	memset(use,0,sizeof(use));
	if(s[0] < 10 || s[0] > 99) return 1;
	n = s[0] % 10;
	use[n] = 1;
	n = s[0] / 10;
	if(use[n]) return 1;
	use[n] = 1;

	for(i = 1 ; i < 3 ; i++){
		t = s[i];
		for(j = 0 ; j < 4 ; j++){
			n = t % 10;
			if(use[n]) return 1;
			use[n] = 1;
			t /= 10;
		}
	}
	return 0;
}

/* ‘½”{’·‰‰Z */
void lmul(int *a,int b,int *c)
{
	int i;
	long d,t,cy = 0L;

	for(i = N - 1 ; i >= 0 ; i--){
		d = a[i];
		t = d * b + cy;
		c[i] = t % 10000;
		cy = t / 10000;
	}
}

/* ƒƒCƒ“ */
int main(void)
{
	int n;
	long s;
	int buf[4],ans[4];

	puts("‚bMAGAZINE PUZZLE");
	for(n = MIN ; n <= MAX ; n++){
		s = (long)n * (n + 1);
		buf[0] = 0;
		buf[1] = s / 10000;
		buf[2] = s % 10000;
		lmul(buf,n + 2,ans);
        if(check(ans) == 0){
			printf("n = %d: %d*%d*%d = %02d %04d %04d\n",
				n,n,n+1,n+2,ans[0],ans[1],ans[2]);
		}
	}

	return 0;
}
